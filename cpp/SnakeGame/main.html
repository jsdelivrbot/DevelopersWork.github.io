#include <iostream>
#include <stdlib.h>
#include <conio.h>
#include <windows.h>
using namespace std;

bool gameOver;

const int x = 40;
const int y = 40;

int foodx,foody,headx,heady;
int score;

enum sdirection{ START = 0, LEFT , RIGHT , UP , DOWN};
sdirection snakedir;

void setup(){
	
	gameOver = false;
	
	snakedir = START;
	
	score = 0;
	
	headx = x/2;
	heady = y/2;
	
	foodx = rand() % x;
	foody = rand() % y;
	
}

void draw(){
	
	system("CLS");
	
	for(int i=0;i<x+3;i++)
		cout<<"=";
	cout<<endl;
	
	for(int j=0;j<x;j++){
		for(int i=0;i<y;i++){
			
		if(i==0 || i == y-1)
			cout<<"||";

		if(j==headx && i==heady)
			cout<<"?";
		else if(foodx == j && foody ==i)
			cout<<"O";
		else cout<<" ";
		
		}
		cout<<endl;
	}
	
	for(int i=0;i<x+3;i++)
		cout<<"=";
	cout<<endl;
	
	cout<<endl<<"\t\tSCORE:"<<score<<endl;

}

void input(){
	
	if(_kbhit()){
		
		switch(_getch()){
			case 'w':
				snakedir = UP;
				break;
			case 's':
				snakedir = DOWN;
				break;
			case 'a':
				snakedir = LEFT;
				break;
			case 'd':
				snakedir = RIGHT;
				break;
			case 'q':
				gameOver = true;
				break;	
		}
	}
	
}

void update(){
	
	switch(snakedir){
		case UP:
			--headx;
			break;
		case DOWN:
			++headx;
			break;
		case LEFT:
			--heady;
			break;
		case RIGHT:
			++heady;
			break;		
	}
	
	if(headx == foodx && foody == heady){
		score++;
		foodx = rand() % x;
		foody = rand() % y;
	}
	
	if( (headx == 0 || heady == 0 ) || (headx >= x  || heady >= y) )
		gameOver = true;
		
	
}



int main(){
	
	setup();
	
	while(!gameOver){
		
		draw();
		
		input();
		
		update();
		
		Sleep(60);
		
	}
	
	if(gameOver)
		cout<<endl<<endl<<"\t\tGAME OVER!";
}
